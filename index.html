Installation & Usage

git clone git@github.com:fs/fs-tool.git ~/.fs-tool

Bash users:

echo 'eval "$($HOME/.fs-tool/bin/fs init -)"' >> ~/.bash_profile
exec bash

Zsh users:

echo 'eval "$($HOME/.fs-tool/bin/fs init -)"' >> ~/.zshenv
source ~/.zshenv
